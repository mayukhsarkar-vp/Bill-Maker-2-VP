function toast(msg,type='info'){
  const el=document.createElement('div');
  el.className=`toast ${type}`;
  const icons={info:'ℹ️',success:'✅',error:'❌'};
  el.innerHTML=`<span>${icons[type]||'ℹ️'}</span><span>${msg}</span>`;

}

function renderBarcode(code){
  const svg=document.createElementNS('http://www.w3.org/2000/svg','svg');
  svg.setAttribute('id','bc-'+code);
  document.body.appendChild(svg);
  try{
    JsBarcode('#bc-'+code,code,{format:'CODE128',width:1.5,height:40,displayValue:true,fontSize:10,margin:4});

    svg.remove();

}

function saveMenuItem(){
  const name=document.getElementById('mi-name').value.trim();
  const cat=document.getElementById('mi-cat').value.trim()||'General';
  const price=parseFloat(document.getElementById('mi-price').value)||0;
  const desc=document.getElementById('mi-desc').value.trim();
  if(!name){toast('Enter item name','error');return;}
  if(editMenuId){
    const i=menuItems.findIndex(m=>m.id===editMenuId);
    if(i>-1)menuItems[i]={...menuItems[i],name,cat,price,desc};
    toast('Item updated','success');
  }else{
    menuItems.push({id:'M'+Date.now(),name,cat,price,desc});
    toast('Item added','success');
  }
  save();clearMenuForm();renderMenuList();
}
function clearMenuForm(){
  editMenuId=null;
  ['mi-name','mi-cat','mi-price','mi-desc','mi-edit-id'].forEach(id=>document.getElementById(id).value='');
  document.getElementById('mf-title').textContent='🌸 Add Menu Item';
  document.getElementById('mi-save-btn').textContent='Add Item';
}
function editMenuItem(id){
  const m=menuItems.find(x=>x.id===id);if(!m)return;
  editMenuId=id;
  document.getElementById('mi-name').value=m.name;
  document.getElementById('mi-cat').value=m.cat;
  document.getElementById('mi-price').value=m.price;
  document.getElementById('mi-desc').value=m.desc||'';
  document.getElementById('mf-title').textContent='✏️ Edit Item';
  document.getElementById('mi-save-btn').textContent='Update Item';
}
function deleteMenuItem(id){
  if(!confirm('Delete this item?'))return;
  menuItems=menuItems.filter(m=>m.id!==id);
  save();renderMenuList();toast('Deleted','info');
}
function renderMenuList(){
  const q=document.getElementById('menu-search').value.toLowerCase();
  const cats=[...new Set(menuItems.map(m=>m.cat))];
  document.getElementById('cat-datalist').innerHTML=cats.map(c=>`<option value="${c}">`).join('');
  const c=document.getElementById('menu-list');
  if(!menuItems.length){c.innerHTML='<div class="empty-state"><div class="es-icon">🍽️</div><p>No items yet. Add your first dish!</p></div>';return;}
  const filtered=menuItems.filter(m=>!q||m.name.toLowerCase().includes(q)||m.cat.toLowerCase().includes(q));
  const grouped={};
  filtered.forEach(m=>{if(!grouped[m.cat])grouped[m.cat]=[];grouped[m.cat].push(m);});
  c.innerHTML=Object.entries(grouped).map(([cat,items])=>`
    <div class="menu-section-title">${cat}</div>
    ${items.map(item=>`
      <div class="menu-item-row">
        <div class="mi-cat">${item.cat}</div>
        <div style="flex:1"><div class="mi-name">${item.name}</div>${item.desc?`<div class="mi-desc">${item.desc}</div>`:''}</div>
        <div class="mi-price">${settings.currency}${item.price.toFixed(2)}</div>
        <button class="btn btn-outline btn-sm btn-icon" onclick="editMenuItem('${item.id}')">✏️</button>
        <button class="btn btn-danger btn-sm btn-icon" onclick="deleteMenuItem('${item.id}')">🗑️</button>
      </div>`).join('')}
  `).join('');
}

function addToBill(id){
  const item=menuItems.find(m=>m.id===id);if(!item)return;
  const ex=billItems.find(b=>b.id===id);
  if(ex)ex.qty++;
  else billItems.push({id:item.id,name:item.name,price:item.price,qty:1});
  renderBillPanel();
}
function renderBillPanel(){
  document.getElementById('bill-order-id').textContent='#'+genOrderId();
  const list=document.getElementById('bill-items-list');
  const totels=document.getElementById('bill-totals');
  if(!billItems.length){
    list.innerHTML='<div class="empty-bill">🍽️<br>No items added yet<br><small>Click menu items to add</small></div>';
    totels.innerHTML='';return;
  }
  list.innerHTML=billItems.map((b,i)=>`
    <div class="bill-row">
      <span class="bill-row-name">${b.name}</span>
      <div class="qty-ctrl">
        <button class="qty-btn" onclick="chQty(${i},-1)">−</button>
        <span class="qty-val">${b.qty}</span>
        <button class="qty-btn" onclick="chQty(${i},1)">+</button>
      </div>

  const t=calcTotals();
  let html=`<div class="total-row"><span>Subtotal</span><span>${settings.currency}${t.sub.toFixed(2)}</span></div>`;
  if(t.dis>0)html+=`<div class="total-row"><span>Discount (${settings.discount}%)</span><span>−${settings.currency}${t.dis.toFixed(2)}</span></div>`;

  if(t.svc>0)html+=`<div class="total-row"><span>Service (${settings.service}%)</span><span>${settings.currency}${t.svc.toFixed(2)}</span></div>`;

function chQty(i,d){billItems[i].qty+=d;if(billItems[i].qty<=0)billItems.splice(i,1);renderBillPanel();}
function removeFromBill(i){billItems.splice(i,1);renderBillPanel();}
function clearBill(){if(billItems.length&&!confirm('Clear bill?'))return;billItems=[];renderBillPanel();}
function calcTotals(){
  const sub=billItems.reduce((s,b)=>s+b.price*b.qty,0);
  const dis=sub*(settings.discount/100);
  const aft=sub-dis;
  const tax=aft*(settings.tax/100);
  const svc=aft*(settings.service/100);
  const total=aft+tax+svc;
  return{sub,dis,tax,svc,total};
}

function saveCustomer(){

  const phone=document.getElementById('cf-phone').value.trim();
  const visits=parseInt(document.getElementById('cf-visits').value)||0;
  const notes=document.getElementById('cf-notes').value.trim();

  if(editCustId){
    const i=customers.findIndex(c=>c.id===editCustId);
    if(i>-1)customers[i]={...customers[i],name,phone,visits,notes};

  }else{
    customers.push({id:'C'+Date.now(),name,phone,visits,notes,joined:new Date().toLocaleDateString('en-IN'),totalSpent:0,lastVisit:new Date().toLocaleDateString('en-IN')});

}

function incrementCustomerVisits(){
  if(!currentBillCustomer)return;
  const i=customers.findIndex(c=>c.id===currentBillCustomer.id);
  if(i>-1){
    customers[i].visits=(customers[i].visits||0)+1;
    customers[i].lastVisit=new Date().toLocaleDateString('en-IN');
    const t=calcTotals();
    customers[i].totalSpent=(customers[i].totalSpent||0)+t.total;
    save();
  }
}

function saveBill(){
  if(!billItems.length){toast('No items','error');return;}
  const t=calcTotals();
  const cust=currentBillCustomer;
  const ordId=genOrderId();
  const now=new Date();
  billHistory.unshift({
    id:ordId,
    date:now.toLocaleDateString('en-IN'),
    time:now.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}),
    customer:cust?cust.name:'Walk-in',
    customerId:cust?cust.id:null,
    phone:cust?cust.phone:'',
    table:document.getElementById('bill-table').value||'—',
    items:[...billItems],
    total:t.total,
    payment:document.getElementById('bill-payment').value,
    status:'saved'
  });
  orderCounter++;
  incrementCustomerVisits();
  save();
  toast(`Bill ${ordId} saved!`,'success');
  billItems=[];renderBillPanel();
  document.getElementById('bill-table').value='';
}

function renderHistory(){
  const q=(document.getElementById('hist-search').value||'').toLowerCase();
  const pay=document.getElementById('hist-filter-pay').value;
  const tbody=document.getElementById('history-tbody');
  if(!billHistory.length){tbody.innerHTML='<tr><td colspan="8" style="text-align:center;padding:40px;color:var(--brown)">No bills saved yet</td></tr>';return;}
  const filtered=billHistory.filter(b=>{
    const qOk=!q||b.id.toLowerCase().includes(q)||b.customer.toLowerCase().includes(q)||(b.phone||'').includes(q);
    const pOk=!pay||b.payment===pay;
    return qOk&&pOk;
  });
  tbody.innerHTML=filtered.map((b,i)=>`
    <tr>
      <td><span style="color:var(--saffron);font-weight:700;font-size:12px">${b.id}</span></td>
      <td>${b.date}<br><span style="font-size:10px;color:var(--brown)">${b.time}</span></td>

      <td>${b.table}</td>
      <td>${b.items.length} item${b.items.length>1?'s':''}</td>
      <td style="color:var(--saffron);font-weight:700">${settings.currency}${b.total.toFixed(2)}</td>
      <td><span class="pay-badge ${b.payment==='cash'?'pay-cash':b.payment==='upi'?'pay-upi':'pay-due'}">${b.payment==='cash'?'💵 Cash':b.payment==='upi'?'📱 UPI':'📋 Due'}</span></td>

}
